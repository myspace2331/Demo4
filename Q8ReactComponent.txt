EXP8: Write a program to demonstrate Class Component & Functional Component using React JS.

Class Component:
Myclass.js file
import React, { Component } from "react";
import logo from '../logo.svg';

class Myclass extends Component {
    show() {
        const list = document.getElementById("myList");
        if (list.style.display === "none") {
            list.style.display = "block";
        } else {
            list.style.display = "none";
        }
    }

    render() {
        return (
            <div>
                <h1>Demo for Class</h1>
                <img src={logo} width='100px' height='100px' alt="logo"/>
                <ul id="myList">
                    <li>About</li>
                    <li>Cources</li>
                    <li>Books</li>
                    <li>Contact us</li>
                </ul>
                <button type="button" onClick={this.show}>ShowHide List</button>
            </div>
        )
    }
}

export default Myclass;
